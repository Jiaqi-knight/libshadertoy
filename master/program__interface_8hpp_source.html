156  resource_interface(const gl::program &program)
157  {
158  // Get the number of active resources
159  GLint active_resources;
160  program.get_program_interface(T::INTERFACE_TYPE, GL_ACTIVE_RESOURCES, &active_resources);
161 
162  // Fetch resource details
163  resources_.reserve(active_resources);
164  for (int resource = 0; resource < active_resources; ++resource)
165  {
166  resources_.emplace_back(program, resource);
167  }
168 
169  // Build lookup maps
170  for (auto it = resources_.begin(); it != resources_.end(); ++it)
171  {
172  size_t idx = it - resources_.begin();
173  resource_names_.insert(std::make_pair(it->name, idx));
174  resource_locations_.insert(std::make_pair(it->location, idx));
175  }
176  }
